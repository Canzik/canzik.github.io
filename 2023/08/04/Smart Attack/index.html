假设在一个曲线E（有限域为）上有点 和 ，要解决，步骤如下：

  1. 将 、两点用 Hensel’s lemma 提升到p的p适域上，并计算其坐标，记为和。在p适域上，x坐标不变，y坐标表示为p进制展开的无穷级数。
  2. 计算$pQ_{Qp}pQ_{Qp}P_tQ_t$。
  3. 计算和。（x()表示点的x坐标，y()表示点的y坐标）
# sagemath
def SmartAttack(P,Q,p):
    E = P.curve()  #获取P点的曲线
    Eqp = EllipticCurve(Qp(p, 2), [ ZZ(t) + randint(0,p)*p for t in E.a_invariants() ])  #Qp()提升到p适域，E.a_invariants()读取E的不变量参数(a, b)

    P_Qps = Eqp.lift_x(ZZ(P.xy()[0]), all=True)  #all=True 返回所有值
    for P_Qp in P_Qps:
        if GF(p)(P_Qp.xy()[1]) == P.xy()[1]:  #验证正确性
            break

    Q_Qps = Eqp.lift_x(ZZ(Q.xy()[0]), all=True)
    for Q_Qp in Q_Qps:
        if GF(p)(Q_Qp.xy()[1]) == Q.xy()[1]:
            break

    p_times_P = p*P_Qp
    p_times_Q = p*Q_Qp

    x_P,y_P = p_times_P.xy()
    x_Q,y_Q = p_times_Q.xy()

    phi_P = -(x_P/y_P)
    phi_Q = -(x_Q/y_Q)
    k = phi_Q/phi_P
    return ZZ(k)
